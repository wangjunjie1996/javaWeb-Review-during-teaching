
1. 使用mybatis做mysql数据库的增、删、改操作
	public int addUser(User u) throws Exception {
		/*
			从数据库连接池获取一个sql连接（SqlSession）
				☆ 事务默认不自动提交：autoCommit = false
		*/
		SqlSession conn = sessionFactory.openSession();

		int ire = conn.insert("test.addUser", u);
		
		//☆ 显示的提交事务，写数据到数据库
		conn.commit();
		
		//释放资源
		conn.close();
		
		return ire;
	}
